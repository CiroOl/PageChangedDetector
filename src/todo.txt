public String showJingleNotificationSelectionDialog(String question) {
		Dialog<String> dialog = new Dialog<>();
		dialog.setTitle("Confirmation");
		dialog.setHeaderText(question);
		
		Button playButton = new Button("â–º");
		playButton.setTextAlignment(TextAlignment.CENTER);
		playButton.setOnAction(event -> {
			AudioClip audio = new AudioClip(getClass().getResource("/music/rush.wav").toExternalForm());
			audio.play();
		});
		
		HBox content = new HBox();
		content.setAlignment(Pos.CENTER);
		Label explenation = new Label("You can only choose default jingle, so far...");
		content.getChildren().addAll(explenation,playButton);
		content.setMargin(playButton, new Insets(2));
		content.setMargin(explenation, new Insets(2));
		dialog.getDialogPane().setContent(content);
		dialog.getDialogPane().getButtonTypes().addAll(ButtonType.OK, ButtonType.CANCEL);
		
		dialog.setResultConverter(dialogButton -> {
			if (dialogButton == ButtonType.OK)
				return "/music/rush.wav";
			return null;
		});
		
		Optional<String> result = dialog.showAndWait();
		if(result.isPresent())
			return result.get();
		else return null;
	}
